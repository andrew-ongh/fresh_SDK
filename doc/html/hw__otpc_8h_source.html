173 static inline void hw_otpc_init(void)
174 {
175  GLOBAL_INT_DISABLE();
176  if ((dg_configBLACK_ORCA_IC_REV == BLACK_ORCA_IC_REV_A)
177  || ((dg_configUSE_AUTO_CHIP_DETECTION == 1) && (CHIP_IS_AE))) {
178  /*
179  * Reset OTP controller
180  */
181  OTPC->OTPC_MODE_REG = HW_OTPC_MODE_STBY; // Set OTPC to standby
182  REG_SET_BIT(CRG_TOP, SYS_CTRL_REG, OTPC_RESET_REQ); // Reset the OTPC
183  REG_CLR_BIT(CRG_TOP, SYS_CTRL_REG, OTPC_RESET_REQ); // Get the OTPC out of Reset
184 
185  /*
186  * Initialize the POR registers
187  */
188  OTPC->OTPC_NWORDS_REG = OTPC_NWORDS_REG_RESET;
189  OTPC->OTPC_TIM1_REG = OTPC_TIM1_REG_RESET;
190  OTPC->OTPC_TIM2_REG = OTPC_TIM2_REG_RESET;
191  }
192 
193  /*
194  * Enable OTPC clock
195  */
196  CRG_TOP->CLK_AMBA_REG |= 1 << REG_POS(CRG_TOP, CLK_AMBA_REG, OTP_ENABLE);
197  GLOBAL_INT_RESTORE();
198 }
