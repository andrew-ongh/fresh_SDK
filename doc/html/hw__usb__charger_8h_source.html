460 __STATIC_INLINE void hw_charger_configure(void)
461 {
462  /* Make sure a rechargeable battery is selected */
463  ASSERT_WARNING( (dg_configBATTERY_TYPE != BATTERY_TYPE_NO_RECHARGE) &&
464  (dg_configBATTERY_TYPE != BATTERY_TYPE_NO_BATTERY) );
465 
466  uint16_t reg;
467 
468 #if (dg_configBATTERY_TYPE == BATTERY_TYPE_2xNIMH)
469  reg = 0x1 << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 3.40V
470 #elif (dg_configBATTERY_TYPE == BATTERY_TYPE_LIFEPO4)
471  reg = 0x3 << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 3.60V
472 #elif (dg_configBATTERY_TYPE == BATTERY_TYPE_LICOO2)
473  reg = 0xA << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 4.20V
474 #elif (dg_configBATTERY_TYPE == BATTERY_TYPE_LIMN2O4)
475  reg = 0xA << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 4.20V
476 #elif (dg_configBATTERY_TYPE == BATTERY_TYPE_NMC)
477  reg = 0xA << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 4.20V
478 #elif (dg_configBATTERY_TYPE == BATTERY_TYPE_LINICOAIO2)
479  reg = 0xA << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 4.20V
480 #elif (dg_configBATTERY_TYPE == BATTERY_TYPE_3xNIMH)
481  reg = 0x11 << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 4.90V
482 #elif (dg_configBATTERY_TYPE == BATTERY_TYPE_CUSTOM)
483  reg = dg_configBATTERY_CHARGE_VOLTAGE << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos;
484 #else
485  reg = 0x0 << ANAMISC_CHARGER_CTRL1_REG_CHARGE_LEVEL_Pos; // 3V
486 #endif
487  reg |= (dg_configBATTERY_CHARGE_NTC << ANAMISC_CHARGER_CTRL1_REG_NTC_DISABLE_Pos);
488  reg |= (dg_configBATTERY_CHARGE_CURRENT << ANAMISC_CHARGER_CTRL1_REG_CHARGE_CUR_Pos);
489  reg |= (2 << ANAMISC_CHARGER_CTRL1_REG_DIE_TEMP_SET_Pos);
490 
491  ANAMISC->CHARGER_CTRL1_REG = reg;
492 
493  sys_tcs_apply(tcs_charger);
494 
495  if (dg_configBATTERY_CHARGE_NTC == 0) {
496  /* Set P14 to output high (3.3V). */
497  GPIO->P1_PADPWR_CTRL_REG &= ~(1 << 4);
498  GPIO->P1_SET_DATA_REG = (1 << 4);
499  GPIO->P14_MODE_REG = 0x300;
500 
501  /* Set P16 to input, no pull-up or pull-down. */
502  GPIO->P16_MODE_REG = 0x0;
503  }
504 }
